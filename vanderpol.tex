\section{Equation}

\begin{equation}
\frac{d^2x}{dt}-\mu (1-x^2) \frac{dx}{dt} + x = 0
\end{equation}
\subsection{Two Dimensional Form}

	Replacing $ \frac{dx}{dt} $ with $ y $ \\
	\begin{equation}
	\frac{dx}{dt} = y
	\end{equation}
    \begin{equation}
	\frac{dy}{dt} = \mu (1-x^2)y -x
	\end{equation}
